void StringAutomaton::S0(const std::string& input) {
    if (input[index] == '\'') {
        inputRead++;
        index++;
        S1(input);
    } else {
        Serr();
    }
}

void StringAutomaton::S1(const std::string& input) {
    if (input[index] == '\'') {
        inputRead++;
    }
    else {
        inputRead++;
        index++;
        S2(input);
    }
}
void StringAutomaton::S2(const std::string& input) {
    while(input[index] != '\''){
        if(input[index] == '\n'){
            newLines++;
        }
        inputRead++;
        index++;
        if(inputRead > input.size()){
            type = TokenType::UNDEFINED;
            std::cout << "ERR";
            break;
        }

    }
    if(input[index] == '\''){
        if(input[index+1] == '\''){
            index++;
            inputRead++;
            S3(input);
        }
        inputRead++;
    } else if(input[index] != '\''){
        Serr();
        inputRead = -1;
    }
}
void StringAutomaton::S3(const std::string& input) {
    // '' but not an empty string
    if(input[index-1] == '\'' && input[index] == '\''){
        index++;
        inputRead++;
        S4(input);
    }
}
